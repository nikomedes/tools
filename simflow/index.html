// cpu_rr.simflow — kein JavaScript nötig
simulation "CPU Round Robin Scheduling" {

  entity Process {
    pid: Int; burst: Int; remaining: Int; wait: Int = 0
    visual: "process"
    visual_states: { "running": { color: "green", pulse: true } }
    label:    "'P' + pid"
    sublabel: "remaining + 'ms'"
  }

  store ready:    Queue<Process>
  store cpu:      Slot<Process>
  store finished: List<Process>

  param quantum:   Int(1..20) = 4  label "Zeitscheibe q (ms)"
  param algorithm: Choice("RR","FIFO","SJF") = "RR"  on_change: "reset"

  // Eingebauter Algorithmus — kein Code nötig:
  use RoundRobin(queue: ready, cpu: cpu, done: finished, quantum: quantum)

  done when ready.empty && cpu.empty

  task "Wartezeit ausrechnen" at done {
    type:      fill_in
    answer:    finished.avg(p => p.wait)
    tolerance: 0.6
  }
}

simulation "LRU Seitenersetzung" {

  entity Page   { number: Int; last_used: Int = 0; visual: "page"   }
  entity Request { number: Int;                        visual: "packet" }

  store frames:   Array<Page>(4)
  store sequence: Queue<Request>
  store log:      List<LogEntry>

  param frame_count: Int(1..8) = 4  label "Anzahl Rahmen"  on_change: "reset"

  // Referenzstring aus Tanenbaum:
  use LRU(frames: frames, requests: sequence, log: log)

  done when sequence.empty

  task "Seitenfehler zählen" at done {
    type:   fill_in
    answer: log.count(e => e.fault)
  }
}

  // Vollständig ohne JavaScript:
